### 代码示例 



 在 js 文件中 

```javascript
Page({
    onShow() {
        const recorderManager = swan.getRecorderManager();
        recorderManager.onError(function(e) {
            swan.showModal({
                title: 'onError success',
                content: JSON.stringify(e)
            });
        });
        const options = {
            duration: 10000,
            sampleRate: 44100,
            numberOfChannels: 1,
            encodeBitRate: 96000,
            format: 'aac'
        };
        recorderManager.start(options);
        this.recorderManager = recorderManager;
    }
});
```
